local pages = {}

local function createPage(name)
    local page = Instance.new("Frame")
    page.Name = name .. "Page"
    page.Size = UDim2.new(1, 0, 1, 0)
    page.BackgroundTransparency = 1
    page.Visible = false
    page.Parent = contentFrame

    local layout = Instance.new("UIListLayout")
    layout.Padding = UDim.new(0, 10)
    layout.SortOrder = Enum.SortOrder.LayoutOrder
    layout.Parent = page

    local padding = Instance.new("UIPadding")
    padding.PaddingTop = UDim.new(0, 14)
    padding.PaddingLeft = UDim.new(0, 14)
    padding.PaddingRight = UDim.new(0, 14)
    padding.PaddingBottom = UDim.new(0, 14)
    padding.Parent = page

    pages[name] = page
    return page
end

local function setPage(name)
    currentPage = name
    for pageName, page in pairs(pages) do
        page.Visible = pageName == name
    end
    for tabName, button in pairs(buttons) do
        button.BackgroundColor3 = tabName == name and Color3.fromRGB(37, 99, 235) or Color3.fromRGB(30, 41, 59)
        button.TextColor3 = tabName == name and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(203, 213, 225)
    end
end

    local button = Instance.new("TextButton")
    button.Name = name .. "Tab"

local function createButton(parent, text, callback)
    local button = Instance.new("TextButton")
    button.Name = text:gsub("%W", "") .. "Button"
    button.Size = UDim2.new(1, 0, 0, 38)
    button.BackgroundColor3 = Color3.fromRGB(37, 99, 235)
    button.BorderSizePixel = 0
    button.AutoButtonColor = false
    button.Font = Enum.Font.GothamSemibold
    button.Text = text
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.TextSize = 13
    button.Parent = parent

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = button

    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.15), {BackgroundColor3 = Color3.fromRGB(59, 130, 246)}):Play()
    end)

    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.15), {BackgroundColor3 = Color3.fromRGB(37, 99, 235)}):Play()
    end)

    button.MouseButton1Click:Connect(function()
        local ok, err = pcall(callback)
        if not ok then
            warn("[StablePanel]", err)
        end
    end)

    return button
end

local function createToggle(parent, text, default, callback)
    local state = default == true
    local button = Instance.new("TextButton")
    button.Name = text:gsub("%W", "") .. "Toggle"
    button.Size = UDim2.new(1, 0, 0, 38)
    button.BackgroundColor3 = state and Color3.fromRGB(22, 163, 74) or Color3.fromRGB(51, 65, 85)
    button.BorderSizePixel = 0
    button.AutoButtonColor = false
    button.Font = Enum.Font.GothamSemibold
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.TextSize = 13
    button.Parent = parent

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = button

    local function render()
        button.Text = text .. ": " .. (state and "ON" or "OFF")
        TweenService:Create(button, TweenInfo.new(0.15), {BackgroundColor3 = state and Color3.fromRGB(22, 163, 74) or Color3.fromRGB(51, 65, 85)}):Play()
    end

    button.MouseButton1Click:Connect(function()
        state = not state
        render()
        local ok, err = pcall(callback, state)
        if not ok then
            warn("[StablePanel]", err)
        end
    end)

    render()
    return button
end

    end

local dragging = false
local dragStart = nil
local startPos = nil

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then
        return
    end
    if input.KeyCode == Enum.KeyCode.RightShift then
        mainFrame.Visible = not mainFrame.Visible
    end
end)

minimizeButton.MouseButton1Click:Connect(function()
    minimized = not minimized
    if minimized then
        tabsFrame.Visible = false
        contentFrame.Visible = false
        version.Visible = false
        mainFrame.Size = UDim2.new(0, 520, 0, 48)
        minimizeButton.Text = "+"
    else
        tabsFrame.Visible = true
        contentFrame.Visible = true
        version.Visible = true
        mainFrame.Size = compactMode and UDim2.new(0, 430, 0, 300) or UDim2.new(0, 520, 0, 360)
        minimizeButton.Text = "-"
    end
end)

closeButton.MouseButton1Click:Connect(function()
    screenGui:Destroy()
end)

RunService.RenderStepped:Connect(function()
    if not screenGui.Parent then
        return
    end
    if performanceMode then
        Lighting.GlobalShadows = false
    end
    if brightMode then
        Lighting.Brightness = 2
        Lighting.ClockTime = 14
    end
end)
